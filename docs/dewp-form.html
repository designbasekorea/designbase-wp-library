<form class="dewp-form">
  <div class="dewp-form-group">
    <label class="dewp-form-label" for="name">이름</label>
    <input id="name" class="dewp-form-input" />
  </div>
  <div class="dewp-form-check dewp-form-checkbox">
    <input class="dewp-form-check-input" type="checkbox" id="c1" />
    <label class="dewp-form-check-label" for="c1">체크</label>
  </div>
</form>
